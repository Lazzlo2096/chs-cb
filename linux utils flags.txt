#Утилита dpkg
#Она для Unix-подобные, POSIX
dpkg -S <something> #Какой-то поиск
dpkg -i <pkg_path> #установить .deb пакет
dpkg -l [маска] #Вывод списка установленных пакетов
dpkg -r <имя_пакета> #Для удаления установленного пакета
